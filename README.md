# chat-bot

> Chat App

## Install

```bash
npm install --save chat-bot
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'chat-bot'
import 'chat-bot/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
